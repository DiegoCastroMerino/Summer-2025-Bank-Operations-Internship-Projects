Sub VerificarDocumentosDeClientesNIU()
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.Calculation = xlCalculationManual
    
    IniciarProceso3
    CrearTablayInsertarFormula3
    CrearPivotTable3
    ConfigurarPivot3
    FormatoCondicional3
    InmovilizarPaneles3

    Application.Calculation = xlCalculationAutomatic
    Application.EnableEvents = True
    Application.ScreenUpdating = True
End Sub

Private Sub IniciarProceso3()
    Dim rowCount As Long
    rowCount = ActiveSheet.Cells(Rows.Count, 1).End(xlUp).Row

    If rowCount > 12000 Then
        MsgBox "Este archivo tiene 12,000 filas o más. Por favor disminuir número de filas"
        End
    End If

    Columns("A:A").Select
    Selection.TextToColumns Destination:=Range("A1"), DataType:=xlDelimited, _
        TextQualifier:=xlDoubleQuote, ConsecutiveDelimiter:=False, Tab:=True, _
        Semicolon:=False, Comma:=False, Space:=False, Other:=True, OtherChar:=";", _
        FieldInfo:=Array(Array(1, 1), Array(2, 1), Array(3, 1), Array(4, 1), Array(5, 1), _
        Array(6, 1), Array(7, 1), Array(8, 1), Array(9, 1), Array(10, 1), Array(11, 1), _
        Array(12, 1), Array(13, 1)), TrailingMinusNumbers:=True

    Application.CutCopyMode = False
    Range("A1").Select
End Sub

Private Sub CrearTablayInsertarFormula3()
    Dim tbl As ListObject
    Dim lastRow As Long, lastCol As Long
    Dim formula365 As String
    Dim formula2016 As String
    Dim formulaToUse As String
    Dim excelMajorVersion As Integer
    Dim targetRange As Range
    Dim ws As Worksheet

    lastRow = Cells(Rows.Count, 1).End(xlUp).Row
    lastCol = Cells(1, Columns.Count).End(xlToLeft).Column

    Set ws = ActiveSheet
    Set tbl = ws.ListObjects.Add(xlSrcRange, Range(Cells(1, 1), Cells(lastRow, lastCol)), , xlYes)
    tbl.Name = "Cuentas1"

    With tbl
        .ListColumns.Add.Name = "Nueva_Lista_NIU"
        Set targetRange = .ListColumns("Nueva_Lista_NIU").DataBodyRange
    End With

    formula365 = "=SI.ERROR(EXTRAE([@[DOC_ID]], ENCONTRAR(""$"", SUSTITUIR([@[DOC_ID]], ""_"", ""$"", LARGO([@[DOC_ID]]) - LARGO(SUSTITUIR([@[DOC_ID]], ""_"", """")))), ENCONTRAR(""2"", [@[DOC_ID]]) - ENCONTRAR(""$"", SUSTITUIR([@[DOC_ID]], ""_"", ""$"", LARGO([@[DOC_ID]]) - LARGO(SUSTITUIR([@[DOC_ID]], ""_"", """"))))), [@[DOC_ID]])"
    formula2016 = formula365

    excelMajorVersion = Val(Application.Version)
    If excelMajorVersion >= 16 Then
        formulaToUse = formula365
    Else
        formulaToUse = formula2016
    End If

    targetRange.FormulaLocal = formulaToUse
    targetRange.Font.Strikethrough = False
End Sub

Private Sub CrearPivotTable3()
    Dim ws As Worksheet

    Set ws = Sheets.Add
    ws.Name = "Tabla_Dinamica"

    ActiveWorkbook.PivotCaches.Create( _
        SourceType:=xlDatabase, _
        SourceData:="Cuentas1" _
    ).CreatePivotTable _
        TableDestination:=ActiveSheet.Cells(1, 1), _
        TableName:="NIU"

    With ws.PivotTables("NIU")
        .ColumnGrand = True
        .HasAutoFormat = True
        .DisplayErrorString = False
        .DisplayNullString = True
        .EnableDrilldown = True
        .ErrorString = ""
        .MergeLabels = False
        .NullString = ""
        .PageFieldOrder = 2
        .PageFieldWrapCount = 0
        .PreserveFormatting = True
        .RowGrand = True
        .SaveData = True
        .PrintTitles = False
        .RepeatItemsOnEachPrintedPage = True
        .TotalsAnnotation = False
        .CompactRowIndent = 1
        .InGridDropZones = False
        .DisplayFieldCaptions = True
        .DisplayMemberPropertyTooltips = False
        .DisplayContextTooltips = True
        .ShowDrillIndicators = True
        .PrintDrillIndicators = False
        .AllowMultipleFilters = False
        .SortUsingCustomLists = True
        .FieldListSortAscending = False
        .ShowValuesRow = False
        .CalculatedMembersInFilters = False
        .RowAxisLayout xlCompactRow
    End With

    With ws.PivotTables("NIU").PivotCache
        .RefreshOnFileOpen = False
        .MissingItemsLimit = xlMissingItemsDefault
    End With
End Sub

Private Sub ConfigurarPivot3()
    ActiveSheet.PivotTables("NIU").RepeatAllLabels xlRepeatLabels

    Dim keepList As Variant
    keepList = Array("_face", "_back", "_front", _
                     "_picture", "PERFIL_TNX.pdf", "Hoja_Comisiones-Banco.pdf", "Cuenta_Ahorro_Digital-Banco.pdf")

    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    Dim item As Variant
    For Each item In keepList
        dict(item) = True
    Next item

    Dim pt As PivotTable
    Set pt = ActiveSheet.PivotTables("NIU")

    With pt
        .ManualUpdate = True
        .RowAxisLayout xlTabularRow
        .RowGrand = True

        With .PivotFields("Nombre")
            .Orientation = xlRowField: .Position = 1
            .Subtotals(1) = False
            .Subtotals = Array(False, False, False, False, False, False, False, False, False, False, False, False)
        End With

        With .PivotFields("NoCuenta")
            .Orientation = xlRowField
            .Position = 2
        End With

        With .PivotFields("TipoProducto")
            .Orientation = xlPageField: .Position = 1
            .ClearAllFilters: .CurrentPage = "CUENTA DIGITAL"
        End With

        With .PivotFields("Nueva_Lista_NIU")
            .Orientation = xlColumnField: .Position = 1
            .ClearAllFilters

            Dim pi As PivotItem
            On Error Resume Next
            For Each pi In .PivotItems
                pi.Visible = dict.exists(pi.Name)
            Next pi
            On Error GoTo 0
        End With

        Dim f As PivotField
        For Each f In .DataFields
            f.Orientation = xlHidden
        Next f

        .AddDataField .PivotFields("Nueva_Lista_NIU"), "Count of Nueva_Lista_NIU", xlCount
        .ManualUpdate = False
    End With

    With ActiveSheet.PivotTables("NIU").PivotFields("Nueva_Lista_NIU")
        .Orientation = xlColumnField
        .Position = 1
    End With
End Sub

Private Sub FormatoCondicional3()
    Dim targetCell As Range
    Dim pt As PivotTable
    Set pt = ActiveSheet.PivotTables("NIU")
    Set targetCell = pt.TableRange1.Cells(3, 3)

    With targetCell
        .FormatConditions.Delete
        .FormatConditions.Add Type:=xlCellValue, Operator:=xlLess, Formula1:="=1"
        With .FormatConditions(.FormatConditions.Count)
            .SetFirstPriority
            .StopIfTrue = False
            .ScopeType = xlDataFieldScope
            With .Interior
                .PatternColorIndex = xlAutomatic
                .ThemeColor = xlThemeColorAccent2
                .TintAndShade = 0.399945066682943
            End With
        End With
    End With
End Sub

Private Sub InmovilizarPaneles3()
    Dim ws3 As Worksheet
    Set ws3 = ThisWorkbook.Sheets("Tabla_Dinamica")

    With ws3.Parent.Windows(1)
        .SplitColumn = 2
        .SplitRow = 4
        .FreezePanes = True
    End With

    With ws3.Columns("B")
        .NumberFormat = "0"
        .AutoFit
    End With
End Sub
